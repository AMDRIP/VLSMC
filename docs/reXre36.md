    asm { "in eax, dx" : in("dx") <= SENSOR_PORT, out("eax") => raw_val }

    idx++;

        // –ú—ã –Ω–∞–∫–æ–ø–∏–ª–∏ 1024 –∑–Ω–∞—á–µ–Ω–∏—è. –ó–∞–ø—É—Å–∫–∞–µ–º –∂–µ—Å—Ç–∫—É—é –º–∞—Ç–µ–º–∞—Ç–∏–∫—É –ø—Ä—è–º–æ –∏–∑ –ø—Ä–µ—Ä—ã–≤–∞–Ω–∏—è!
        // –î–µ–∫–æ—Ä–∞—Ç–æ—Ä fast –æ—Ç–∫–ª—é—á–∞–µ—Ç —Å—Ç—Ä–æ–≥—É—é –ø—Ä–æ–≤–µ—Ä–∫—É NaN/Inf –¥–ª—è –º–∞–∫—Å–∏–º–∞–ª—å–Ω–æ–π —Å–∫–æ—Ä–æ—Å—Ç–∏
        fast {

            }
        }
        idx = 0;
    }
    
    hardware::ack_interrupt();
}
